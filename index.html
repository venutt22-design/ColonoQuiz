<script>

/* ============================================================

   ⚙️  CONFIG — แก้ค่าเหล่านี้ก่อนใช้งาน

const CONFIG = {

  WEBAPP_URL:      AKfycbzVOliZZp8wcSwqw0hzZXfzFFJ8IzD_CcdRO27H7nmyTDGzAe6wjDpQj6v-4Epas_s8gQ,  // URL จาก Deploy > Web App

  PASS_THRESHOLD:  0.8,                                   // เกณฑ์ผ่าน (0.8 = 80%)

  LINK_MAIN_MENU:  https://venutt22-design.github.io/COLONONEWAPP/checklist,        // ลิงก์กลับเมนูหลัก

  LINK_LESSON:     https://venutt22-design.github.io/COLONONEWAPP/athome,           // ลิงก์กลับบทเรียน

};


/* ============================================================

   STATE

============================================================ */

let questions = [], shuffled = [], idx = 0, score = 0;

let answers = [], userName = "", startTime = null;


/* ============================================================

   SCREENS

============================================================ */

const screens = {

  start:  document.getElementById("s-start"),

  loadq:  document.getElementById("s-loadq"),

  quiz:   document.getElementById("s-quiz"),

  saving: document.getElementById("s-saving"),

  result: document.getElementById("s-result"),

};

function show(name) {

  Object.values(screens).forEach(s => s.classList.remove("active"));

  screens[name].classList.add("active");

  window.scrollTo({ top: 0, behavior: "smooth" });

}


/* ============================================================

   START

============================================================ */

document.getElementById("btn-start").addEventListener("click", handleStart);

document.getElementById("name-inp").addEventListener("keyup", e => { if (e.key === "Enter") handleStart(); });


async function handleStart() {

  const name = document.getElementById("name-inp").value.trim();

  if (!name) { flashBorder(document.getElementById("name-inp")); return; }

  userName = name;


  const btn = document.getElementById("btn-start");

  btn.disabled = true;

  btn.textContent = "กำลังโหลด...";

  document.getElementById("start-error").classList.add("hidden");


  show("loadq");


  try {

    questions = await fetchQuestions();

  } catch (err) {

    show("start");

    btn.disabled = false;

    btn.textContent = "เริ่มทำแบบทดสอบ";

    showErr("ไม่สามารถโหลดคำถามได้ กรุณาตรวจสอบการเชื่อมต่ออินเทอร์เน็ตแล้วลองใหม่อีกครั้ง");

    return;

  }


  if (!questions || questions.length === 0) {

    show("start");

    btn.disabled = false;

    btn.textContent = "เริ่มทำแบบทดสอบ";

    showErr("ไม่พบคำถามใน Google Sheets กรุณาเพิ่มคำถามก่อน");

    return;

  }


  idx = 0; score = 0; answers = [];

  shuffled  = shuffle([...questions]);

  startTime = Date.now();

  show("quiz");

  renderQuestion();

}


function showErr(msg) {

  const el = document.getElementById("start-error");

  el.textContent = msg;

  el.classList.remove("hidden");

}


/* ============================================================

   FETCH QUESTIONS  (GET ?action=getQuestions)

============================================================ */

async function fetchQuestions() {

  const res  = await fetch(CONFIG.WEBAPP_URL + "?action=getQuestions");

  if (!res.ok) throw new Error("HTTP " + res.status);

  const data = await res.json();

  if (data.error) throw new Error(data.error);

  // data.questions = [{ q:"...", c:["a","b","c","d"], a:"..." }, ...]

  return data.questions;

}


/* ============================================================

   QUIZ

============================================================ */

const LETTERS = ["ก", "ข", "ค", "ง", "จ"];


function renderQuestion() {

  const total = shuffled.length;

  const q     = shuffled[idx];

  const pct   = Math.round((idx / total) * 100);


  document.getElementById("q-label").textContent      = `คำถามที่ ${idx + 1} / ${total}`;

  document.getElementById("q-score-live").textContent  = `คะแนน ${score}`;

  document.getElementById("q-bar").style.width         = pct + "%";

  document.getElementById("q-text").textContent        = q.q;


  const choicesEl = document.getElementById("q-choices");

  choicesEl.innerHTML = "";

  const opts = shuffle([...q.c]);

  opts.forEach((opt, i) => {

    const div = document.createElement("div");

    div.className = "choice";

    div.innerHTML = `<span class="choice-key">${LETTERS[i]}</span><span class="choice-text">${opt}</span>`;

    div.addEventListener("click", () => pick(opt, div, q.a));

    choicesEl.appendChild(div);

  });

}


function pick(chosen, el, correct) {

  document.querySelectorAll(".choice").forEach(c => c.classList.add("disabled"));

  el.classList.add("selected");

  const isCorrect = chosen === correct;

  if (isCorrect) score++;

  answers.push({ q: shuffled[idx].q, chosen, correct, isCorrect });

  setTimeout(() => { idx++; idx < shuffled.length ? renderQuestion() : finish(); }, 500);

}


/* ============================================================

   FINISH & SAVE  (POST action=saveResult)

============================================================ */

async function finish() {

  const timeSec = ((Date.now() - startTime) / 1000).toFixed(1);

  show("saving");


  document.getElementById("btn-home").href   = CONFIG.LINK_MAIN_MENU;

  document.getElementById("btn-lesson").href = CONFIG.LINK_LESSON;


  const payload = {

    action:   "saveResult",

    name:     userName,

    score:    score,

    total:    shuffled.length,

    percent:  Math.round((score / shuffled.length) * 100),

    passed:   (score / shuffled.length) >= CONFIG.PASS_THRESHOLD,

    timeSec:  timeSec,

    datetime: new Date().toLocaleString("th-TH", { timeZone: "Asia/Bangkok" }),

  };


  try {

    await fetch(CONFIG.WEBAPP_URL, {

      method:   "POST",

      headers:  { "Content-Type": "text/plain;charset=utf-8" },

      body:     JSON.stringify(payload),

      redirect: "follow",

    });

  } catch (err) {

    console.warn("บันทึกไม่สำเร็จ:", err);

  }


  showResult(score, shuffled.length);

}


/* ============================================================

   RESULT

============================================================ */

function showResult(score, total) {

  const ratio  = score / total;

  const passed = ratio >= CONFIG.PASS_THRESHOLD;

  document.getElementById("r-name").textContent = `ผู้เข้าทำแบบทดสอบ: ${userName}`;


  const box = document.getElementById("r-box");

  if (passed) {

    box.innerHTML = `

      <div class="result-wrap pass">

        <div class="result-emoji">🎉</div>

        <div class="result-status">ยินดีด้วย! คุณผ่านแบบทดสอบ</div>

        <div class="result-score">คะแนน ${score} / ${total} (${Math.round(ratio*100)}%)</div>

        <div class="result-desc">คุณมีความเข้าใจในการเตรียมลำไส้ก่อนการส่องกล้องเป็นอย่างดี<br>สามารถดำเนินการตามขั้นตอนได้อย่างถูกต้อง</div>

      </div>`;

    document.getElementById("r-pass-actions").classList.remove("hidden");

    document.getElementById("r-fail-actions").classList.add("hidden");

  } else {

    box.innerHTML = `

      <div class="result-wrap fail">

        <div class="result-emoji">📖</div>

        <div class="result-status">ยังไม่ผ่านแบบทดสอบ</div>

        <div class="result-score">คะแนน ${score} / ${total} (${Math.round(ratio*100)}%)</div>

        <div class="result-desc">เกณฑ์ผ่านคือร้อยละ ${Math.round(CONFIG.PASS_THRESHOLD*100)} ขึ้นไป<br>แนะนำให้กลับไปทบทวนบทเรียนอีกครั้ง</div>

      </div>`;

    document.getElementById("r-fail-actions").classList.remove("hidden");

    document.getElementById("r-pass-actions").classList.add("hidden");

  }

  show("result");

}


document.getElementById("btn-retry").addEventListener("click", () => {

  document.getElementById("name-inp").value = "";

  document.getElementById("btn-start").disabled = false;

  document.getElementById("btn-start").textContent = "เริ่มทำแบบทดสอบ";

  show("start");

});


/* ============================================================

   HELPERS

============================================================ */

function shuffle(arr) {

  for (let i = arr.length - 1; i > 0; i--) {

    const j = Math.floor(Math.random() * (i + 1));

    [arr[i], arr[j]] = [arr[j], arr[i]];

  }

  return arr;

}

function flashBorder(el) {

  el.style.borderColor = "#e74c3c"; el.focus();

  setTimeout(() => { el.style.borderColor = ""; }, 1500);

}
